#include<bits/stdc++.h>
#define int long long
using namespace std;
int n,m,ans;
double con[1500][1500];
struct node{
	double x,y;
} p[1500];
double dis(int a,int b)
{
	return sqrt((p[a].y-p[b].y)*(p[a].y-p[b].y)+(p[a].x-p[b].x)*(p[a].x-p[b].x));
}
signed main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++)
		cin>>p[i].x>>p[i].y;
	for(int i=1;i<=n;i++)
		for(int l=1;l<=n;l++)
			if(i!=l)
				con[i][l]=1e9;
	while(m--)
	{
		int a,b;
		cin>>a>>b;
		con[a][b]=con[b][a]=dis(a,b);
	}
	for(int k=1;k<=n;k++)
		for(int i=1;i<=n;i++)
			for(int l=1;l<=n;l++)
			{
				con[i][l]=min(con[i][l],con[i][k]+con[k][l]);
				con[l][i]=con[i][l];
			}
	for(int i=1;i<=n-2;i++)
		for(int l=i+1;l<=n-1;l++)
			for(int j=l+1;j<=n;j++)
			{
				if(con[i][l]==dis(i,l)&&con[i][j]==dis(i,j)&&con[l][j]==dis(l,j))
				{
					double _max=max(dis(i,l),max(dis(l,j),dis(i,j)));
					if(dis(i,l)+dis(l,j)+dis(i,j)-_max>_max)
						ans++;
				}
			}	
	cout<<ans;
	return 0;
}

看到 kth 很容易想到主席树，用主席树维护即可，每一次加入的是一段区间，注意细节和空间问题。

时间复杂度 O(n\log^2n)， std最慢的点是450ms 。

#include <queue>
#include <cstdio>
#include <string>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;

const int N=1e5+5;
const int M=2e5+5;
const int mod=998244353;

ll kth[M];
int sum[N],inv[N],power[N];
int n,m,q,S[M],T[M],a[N];

namespace IO {
    inline int mul(int x,int y) {return 1ll*x*y%mod;}
    inline int dec(int x,int y) {x-=y;if(x<0) x+=mod;return x;}
    inline int add(int x,int y) {x+=y;if(x>=mod) x-=mod;return x;}
    inline void inc(int&x,int y) {x+=y;if(x>=mod) x-=mod;}
    inline int modpow(int x,int y,int res=1) {
        for(;y;y>>=1,x=mul(x,x)) if(y&1) res=mul(res,x);
        return res;
    }
    template <typename _Tp> inline void IN(_Tp&x) {
        char ch;bool flag=0;x=0;
        while(ch=getchar(),!isdigit(ch)) if(ch=='-')  flag=1;
        while(isdigit(ch)) x=x*10+ch-'0',ch=getchar();
        if(flag) x=-x;
    }
}using namespace IO;

int rt[M];
struct Segment_Tree {
    #define mid ((l+r)>>1)
    int cnt=0;
    struct Node {int lc,rc,tag;ll val;} t[N*70];
    inline void Insert(int&now,int last,int l,int r,int L,int R) { /*sum-insert*/
        now=++cnt,t[now]=t[last],t[now].val+=(R-L+1);
        if(L==l&&r==R) {t[now].tag++;return;}
        if(R<=mid) Insert(t[now].lc,t[last].lc,l,mid,L,R);
        else if(L>mid) Insert(t[now].rc,t[last].rc,mid+1,r,L,R);
        else {
            Insert(t[now].lc,t[last].lc,l,mid,L,mid);
            Insert(t[now].rc,t[last].rc,mid+1,r,mid+1,R);
        }
    }
    inline int query_kth(int now,int last,ll tag1,ll tag2,int l,int r,ll k) {/*kth-query*/
        if(l==r) return l;
        tag1+=t[now].tag,tag2+=t[last].tag;
        ll th=1ll*tag2*(mid-l+1)+t[t[last].lc].val-1ll*tag1*(mid-l+1)-t[t[now].lc].val;
        if(k<=th) return query_kth(t[now].lc,t[last].lc,tag1,tag2,l,mid,k);
        else return query_kth(t[now].rc,t[last].rc,tag1,tag2,mid+1,r,k-th);
    }
    inline ll query_tot(int now,int last,ll tag,int l,int r,int pos) {/*tot-query*/
        tag+=t[last].tag-t[now].tag;
        if(l==r) return tag;
        if(pos<=mid) return query_tot(t[now].lc,t[last].lc,tag,l,mid,pos);
        else return query_tot(t[now].rc,t[last].rc,tag,mid+1,r,pos);
    }
    inline int query_sum(int now,int last,int l,int r) { /*sum-query*/
        if(t[last].tag-t[now].tag) return dec(sum[r],sum[l-1]);
        int res=0;
        if(t[last].lc) inc(res,query_sum(t[now].lc,t[last].lc,l,mid));
        if(t[last].rc) inc(res,query_sum(t[now].rc,t[last].rc,mid+1,r));
        return res;
    }
    inline int query_mul(int now,int last,int l,int r) { /*sum-query*/
        if(t[last].tag-t[now].tag) return mul(power[r],inv[l-1]);
        int res=1,tmp=0;
        if(t[last].lc)
            tmp=query_mul(t[now].lc,t[last].lc,l,mid),
            res=mul(res,tmp==0?1:tmp),tmp=0;
        if(t[last].rc)
            tmp=query_mul(t[now].rc,t[last].rc,mid+1,r),
            res=mul(res,tmp==0?1:tmp),tmp=0;
        return res;
    }
}tree;

char opt[N];
int main() {
    IN(n),IN(m),IN(q);
    for(int i=1;i<=n;++i) IN(a[i]);
    power[0]=inv[0]=1;
    for(int i=1;i<=n;++i) sum[i]=add(sum[i-1],a[i]);
    for(int i=1;i<=n;++i) power[i]=mul(power[i-1],a[i]),inv[i]=modpow(power[i],mod-2);
    for(int i=1;i<=m;++i) IN(S[i]),IN(T[i]),kth[i]=kth[i-1]+T[i]-S[i]+1;
    for(int i=1;i<=m;++i) tree.Insert(rt[i],rt[i-1],1,n,S[i],T[i]);
    while(q--) {
        scanf("%s",opt);
        int l,r,k,x;IN(l),IN(r);
        if(opt[0]=='q') {
            if(opt[6]=='k') {
                IN(k);
                if(kth[r]-kth[l-1]<k) puts("0");
                else printf("%d\n",tree.query_kth(rt[l-1],rt[r],0,0,1,n,k));
            } else if(opt[6]=='t') {
                IN(x),
                printf("%lld\n",tree.query_tot(rt[l-1],rt[r],0,1,n,x));
            } else if(opt[6]=='s') {
                printf("%d\n",tree.query_sum(rt[l-1],rt[r],1,n));
            } else if(opt[6]=='m') {
                if(kth[r]-kth[l-1]==0) puts("0");
                else printf("%d\n",tree.query_mul(rt[l-1],rt[r],1,n));
            }
        } else {
            ++m,S[m]=l,T[m]=r,kth[m]=kth[m-1]+T[m]-S[m]+1;
            tree.Insert(rt[m],rt[m-1],1,n,S[m],T[m]);
        }
    }
    return 0;
}
